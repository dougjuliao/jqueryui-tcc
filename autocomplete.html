Autocomplete

Esse efeito é usado para auto completar campos que estão sendo digitados.

O código para o default seria assim: 


                $(function() {
                    var availableTags = [ //Textos que irão se auto completar.
                        "ActionScript",
                        "AppleScript",
                        "Asp",
                        "BASIC",
                        "C",
                        "C++",
                        "Clojure",
                        "COBOL",
                        "ColdFusion",
                        "Erlang",
                        "Fortran",
                        "Groovy",
                        "Haskell",
                        "Java",
                        "JavaScript",
                        "Lisp",
                        "Perl",
                        "PHP",
                        "Python",
                        "Ruby",
                        "Scala",
                        "Scheme"
                    ];
                    $( "#tags" ).autocomplete({
                        source: availableTags
                    });
                });
